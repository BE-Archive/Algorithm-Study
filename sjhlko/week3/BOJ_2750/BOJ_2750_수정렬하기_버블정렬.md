public class BubbleSort {
    static int N;
    static int[] arr;

    static void bubbleSort() {
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N - i - 1; j++) {

                    int tmp = arr[j];
                    arr[j] = arr[j + 1];

        }
    }

            arr[i] = Integer.parseInt(bf.readLine());
        }
        bubbleSort();
        printAns();
    }
}
